== Nachhaltigkeitstracker - UML-Diagramme

=== Datenmodell (UML-Klassendiagramm)

[plantuml]
----
@startuml
class Produkt {
    - ID: int
    - Name: String
    - Hersteller: String
    - Herkunftsland: String
    - Nachhaltigkeitsscore: float
    - KategorieID
    - BarcodeID: int
}

class ProduktKategorie{
    - ID: int
    - Bezeichnung: string
}

class Barcode {
    - ID: int
    - Code: String
}

class Nutzer {
    - ID: int
    - Name: String
    - Email: String
    - TelNr: String
}

Produkt <--  Barcode
Produkt <-- ProduktKategorie
@enduml
